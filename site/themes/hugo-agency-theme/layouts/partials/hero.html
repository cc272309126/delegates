Quickstart

$ wget http://bit.ly/simple_delegates -O delegate.h

$ cat <<EOT >> test.cpp

#include "delegate.h"

#include "stdio.h"

struct listener {

    void foo(int x) { printf("got %d\n", x);} };

int main() {

    listener l;

    auto d = dlgt::make_delegate(&listener::foo, l);

    d(42); return 0; }

EOT

$ g++ test.cpp --std=c++11 -o test && ./test

got 42
